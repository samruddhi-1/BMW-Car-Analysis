# create budget ranges using price quantile
In[19]:
q1,q2 = df['price'].quantile([0.33 , 0.66])

def budget_range(price):
    if price <= q1:
        return "Low Budget"
    elif price <= q2:
        return "Mid Budget"
    else:
        return "High Budget"

df['budget_range'] = df['price'].apply(budget_range)
In[22]:
df['value_score'] =df['performance'] / df['price']
