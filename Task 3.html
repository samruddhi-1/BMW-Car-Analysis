#calculate yearly Depriciation Per Model

depreciation_results = []
for model in model_age_price['model'].unique():
    data = model_age_price[model_age_price['model'] == model].sort_values('Age')

    data['yearly_dep'] = data['price'].shift(1) - data['price']

    data['dep_%'] = (data['yearly_dep'] / data['price'].shift(1)) * 100

    depreciation_results.append(data)
depreciation_df = pd.concat(depreciation_results)
